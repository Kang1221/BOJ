#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

//for 오름차순 정렬
bool cmp(int a, int b) { return a > b; }

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int n, tmp;
	vector<int> C;

	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> tmp;
		C.push_back(tmp);
	}

	//vector 오름차순 정렬
	sort(C.begin(), C.end(), cmp);

	int j=0, num_a=1, b = C[0] ;
	vector<int> Num(n,0);

	for (int i = 0; i < n; i++) {
		if (b != C[i]) { j++; }

		Num[j]++;
		b = C[i];
	}
	
	sort(Num.begin(), Num.end(), cmp);
	cout << Num[0];

	return 0;
}
