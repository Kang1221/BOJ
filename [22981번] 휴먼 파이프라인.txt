//20%까지 돌다가 틀렸습니다. 

#include <iostream>
#include <vector>
#include <algorithm>
#include<cmath>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    long long n, tmp;
    long long v_sum=0, max = 0;
    double k;
    vector <long long> v;

    cin >> n >> k;
    for (int i = 0; i < n; i++) {
        cin >> tmp;
        v.push_back(tmp);
    }

    //정렬
    sort(v.begin(), v.end());
    
    for (int i = 1; i < n; i++) {
        v_sum = v[0] * i + v[i] * (n - i);
        if (v_sum > max)
            max = v_sum;
    }

    cout << ceil(k / max);

    return 0;
}
